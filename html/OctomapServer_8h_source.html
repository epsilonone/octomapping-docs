123  inline bool isInUpdateBBX(const OcTreeT::iterator& it) const {
124  // 2^(tree_depth-depth) voxels wide:
125  unsigned voxelWidth = (1 << (m_maxTreeDepth - it.getDepth()));
126  octomap::OcTreeKey key = it.getIndexKey(); // lower corner of voxel
127  return (key[0] + voxelWidth >= m_updateBBXMin[0]
128  && key[1] + voxelWidth >= m_updateBBXMin[1]
129  && key[0] <= m_updateBBXMax[0]
130  && key[1] <= m_updateBBXMax[1]);
131  }
